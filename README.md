# Installation:

# Getting Started:

run btcd: btcd -u rpcuser -P rpcpass (--testnet --mainnet)

create btcwallet: btcwallet -u rpcuser -P rpcpass --create

run bctwallet: btcwallet -u rpcuser -P rpcpass -d trace

run server.go
